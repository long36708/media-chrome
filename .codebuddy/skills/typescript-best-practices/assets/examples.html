// 字面量类型示例
const UserRole = {
  ADMIN: 'admin',
  USER: 'user',
  GUEST: 'guest',
} as const;

type UserRoleType = (typeof UserRole)[keyof typeof UserRole];
// UserRoleType = 'admin' | 'user' | 'guest'

// 类型守卫示例
function isString(value: unknown): value is string {
  return typeof value === 'string';
}

if (isString(value)) {
  // TypeScript 知道 value 是 string
}

// 类型推断示例
function identity(arg: T): T {
  return arg;
}

const result = identity('hello'); // 推断为 string

泛型示例

泛型函数
// 泛型函数示例
function map(array: T[], fn: (item: T) => U): U[] {
  return array.map(fn);
}

const numbers = [1, 2, 3, 4, 5];
const doubled = map(numbers, n => n * 2);
// doubled: number[]

function filter(array: T[], predicate: (item: T) => boolean): T[] {
  return array.filter(predicate);
}

const evens = filter(numbers, n => n % 2 === 0);
// evens: number[]

工具类型示例

Partial 类型
Pick 类型
Omit 类型
// 工具类型示例
interface User {
  id: number;
  name: string;
  email: string;
  age: number;
}

// Partial - 使所有属性可选
type PartialUser = Partial;
// { id?: number; name?: string; email?: string; age?: number; }

// Pick - 选择特定属性
type UserContact = Pick;
// { name: string; email: string; }

// Omit - 排除特定属性
type UserWithoutId = Omit;
// { name: string; email: string; age: number; }

// Readonly - 使所有属性只读
type ReadonlyUser = Readonly;
// { readonly id: number; readonly name: string; ... }

// 条件类型示例
type NonNullable = T extends null | undefined ? never : T;

type WithId = T extends { id?: number }
  ? { id: number } & T
  : T;

type StatusResult =
  | { status: 'success'; data: T }
  | { status: 'error'; error: E }
  | { status: 'loading' };

// 使用示例

类型安全示例

类型安全检查
// 类型安全示例
// 使用 unknown 而不是 any
function parseJSON(json: string): unknown {
  return JSON.parse(json);
}

const data = parseJSON('{"name":"John"}');

// 必须在使用前检查类型
if (typeof data === 'object' && data !== null && 'name' in data) {
  const name = (data as { name: string }).name;
  console.log(name);
}

// 使用类型守卫
function isUser(value: unknown): value is { name: string; email: string } {
  return typeof value === 'object' &&
         value !== null &&
         'name' in value &&
         'email' in value;
}

if (isUser(data)) {
  console.log(data.name, data.email);
}
